\subsection*{Derivational Relations}
\begin{frame}{Derivational Relations}

\begin{itemize}
\item Often words are linked by more or less systematic relations, sometimes morphologically marked
  \begin{itemize}
  \item \lex{beauty}/\lex{beautiful}/\lex{beautify}
  \item \lex{cute}/\lex{cuteness}
  \end{itemize}
\item These can be very productive, but can also change the meaning of the derived word
\end{itemize}
\end{frame}

\begin{frame}{Agentive Nouns}
\MyLogo{}
\begin{itemize}
\item An \txx{agentive noun} is a word that is typically derived from
  another word denoting an action, and that identifies an entity that
  does that action.  \\ \textbf{verb} + \lex{-er, -or, -ant}
  \begin{exe}
    \ex \lex{murderer, commentator, whaler, director, computer}
    \ex ?? \lex{undertaker, cooker, footballer} (Saeed also includes these)
  \end{exe}
\item Should \lex{murderer} be listed separately from \lex{murder} in
the dictionary? Why or why not?
\item Also \txx{recipient noun}s that show the undergoer:
  \textbf{verb} + \lex{-ee}: \lex{employee}, \lex{trustee}
\end{itemize}
\end{frame}


\begin{frame}[allowframebreaks]{Agentive Nouns in Other Languages}

  

\begin{itemize}
\item Slovak 
  \begin{itemize}
  \item \lex{-ník}/\lex{-níčka}: more concrete or traditional
      \begin{exe}
        \ex \lex{pracovať} (to work) - \lex{pracovník} (m.), \lex{pracovníčka} (f.)
        \ex \lex{účtovať} (to do accounting) - \lex{účtovník} (m.), \lex{účtovníčka} (f.)
        \ex \lex{poľovať} (to hunt) - \lex{poľovník} (m.), \lex{poľovníčka} (f.)
      \end{exe}
    \item \lex{-teľ}/\lex{-teľka}
      \begin{exe}
        \ex \lex{učiť} (to teach) - \lex{učiteľ} (m.), \lex{učiteľka} (f.)
        \ex \lex{organizovať} (to manage) - \lex{organizovateľ} (m.), \lex{organizovateľka}  (f.)
        \ex \lex{riadiť} (to lead) - \lex{riaditeľ} (m.), \lex{riaditeľka} (f.)
      \end{exe}
      $^*$the word riaditeľ does not mean a leader but it is used in a school for headmaster 
    \item  \lex{-ár}/\lex{árka}: this suffix is used for someone who works with/produces/deals with something
      \begin{exe}
        \ex \lex{ piecť} (to bake) - \lex{pekár} (m.), \lex{pekárka} (f.)
        \ex \lex{ryba} (fish) - \lex{rybár} (m.), \lex{rybárka} (f.)
        \ex \lex{víno} (wine) - \lex{vinár} (m.), \lex{vinárka} (f.)
      \end{exe}
    \item \lex{-ista}/{-istka} this suffix is used for borrowed/lent words; an "international suffix"
      \begin{exe}
        \ex \lex{futbal} (noun: football) - \lex{futbalista} (m.), \lex{futbalistka} (f.)
        \ex \lex{basketbal} (noun: basketball) - \lex{basketbalista} (m.), \lex{basketbalistka} (f.)
        \ex \lex{volejbal} (noun: volleyball) - \lex{volejbalista} (m.), \lex{volejbalistka} (f.)
      \end{exe}
  \end{itemize}
  Thanks to Emma Martincova for the Slovak examples
  \framebreak
\item Japanese (suffix distinguishes person/machine)
  
  \begin{itemize}
  \item \jpf{運転する} → \jpf{運転者} \eng[driver]{unten-sha}
  \item \jpf{計算する} → \jpf{計算者} \jpf{計算機} \eng[computer]{keisan-sha/ki}
  \item \jpf{研究する} → \jpf{研究者} \jpf{研究員} \eng[researcher]{kenkyuu-sha/in}
  \item \jpf{読む} → \jpf{読み手} \jpf{読者} \jpn[reader]{yomite/dokusha}
  \end{itemize}
\item Malay (prefix can convert any part  of speech)
  \begin{itemize}
  \item \eng[help]{bantu (v)} → \eng[assistant/helper]{pembantu} 
  \item \eng[cut]{potong (v)} → \eng[cutter (human/machine)]{pemotong}
  \item \eng[fly]{terbang (v)} → \eng[pilot (not passenger)]{penerbang}
%  \item \zsm[scissors]{gunting (n)} → \zsm[(editor – human)]{penyunting}
  \end{itemize}

\end{itemize}
\end{frame}
