\begin{frame}{Color Terms}
\newcommand{\COL}[1]{\textcolor{#1}{\textbf{#1}}}

\begin{itemize}
\MyLogo{See also: \url{http://wals.info/chapter/133}: Colour Terms
by Paul Kay and Luisa Maffi}
\item \txx{Basic Color Terms}
  \begin{itemize}
  \item Monolexemic
  \item Not a hyponym of any other color
  \item Can be widely applied
  \item Not derived from a noun
  \end{itemize}
\item \txx{Focal Colors} are related to the neurophysiology of our visual system
\item Seem to come in an order \\

\hspace*{-3em}\makebox{$ \left\{ \begin{array}[c]{l}
    \mbox{\textbf{white/dark}} \\  \mbox{\textbf{black/light}}
  \end{array} \right\}
   < \mbox{\COL{red}} < 
 \left\{ \begin{array}[c]{l}
    \mbox{\COL{green}} \\  \mbox{\COL{yellow}}
  \end{array} \right\}
 < \mbox{\COL{blue}} < \mbox{\COL{brown}} <  
\left\{ \begin{array}[c]{l}
    \mbox{\COL{purple}} \\  \mbox{\COL{pink}} \\ \mbox{\COL{orange}} \\  \mbox{\COL{grey}}
  \end{array} \right\} $}
\end{itemize}

Note: \COL{pink} is really light \COL{red}, and  \COL{orange} is light \COL{brown}, \ldots

\end{frame}

\begin{frame}[allowframebreaks]{Core Vocabulary}
\MyLogo{\url{en.wiktionary.org/wiki/Appendix:Swadesh_list}}
\begin{itemize}
\item Some universal terms can be used to compare languages
  \begin{itemize}
  \item lexicostatistics (quantitative language relatedness assessment)
  \item glottochronology (language divergence dating)
\end{itemize}
\item The \txx{Swadesh list}, developed by Morris Swadesh from 1940
  onward 
\item Chosen for their universal, culturally independent availability in as many languages as possible
\end{itemize}
